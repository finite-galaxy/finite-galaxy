mission "Bounty Hunting (Big)"
  name "Hunt down the <npc>"
  description "A pirate vessel named the <npc> has been attacking merchants near the <system> system. Destroy it and return to <planet> for payment (<payment>)."
  repeat
  job
  to offer
    "combat rating" > 500
    random < 20
  source
    government "Deep" "Free Worlds" "Neutral" "Republic" "Corporation"
    attributes core "dirt belt" frontier north rim south
  npc kill
    government "Bounty"
    personality heroic staying nemesis target
    system
      distance 1 2
    fleet
      names "pirate"
      variant 1
        "Lance"
      variant 1
        "Leviathan"
      variant 1
        "Vanguard"
    dialogue "The <npc> has been eliminated. You can claim the bounty payment by returning to <destination>."
  on complete
    payment 250000
    dialogue "The government of <planet> gratefully pays you <payment> for eliminating the <npc>."
